    function parseItalianNumber(value) {

      if (!value) return NaN;

      let cleaned = String(value).trim().split(' ').join('');

      if (cleaned.indexOf(',') !== -1 && cleaned.indexOf('.') !== -1) {

        cleaned = cleaned.split('.').join('').replace(',', '.');

      } else if (cleaned.indexOf(',') !== -1) {

        cleaned = cleaned.replace(',', '.');

      } else if (cleaned.indexOf('.') !== -1) {

        const parts = cleaned.split('.');

        if (parts.length > 1 && parts[parts.length - 1].length === 3) cleaned = parts.join('');

      }

      return Number(cleaned);

    }


    function formatThousandsInput(value) {

      const digitsOnly = String(value || '').replace(/[^0-9]/g, '');

      if (!digitsOnly) return '';

      return digitsOnly.replace(/\B(?=(\d{3})+(?!\d))/g, '.');

    }


    function formatNumber(value, decimals) {

        showError('Indica se hai bollette alte per potenza o picchi di assorbimento.');

        return;

      }

      if (fvSelection.value === 'si' && (!isFinite(fvPower) || fvPower <= 0)) {

        showError('Inserisci una potenza fotovoltaica valida.');

        return;

      }


      if (fvSelection.value === 'no') {

        setResult('resBattery', 'Da valutare con FV');

        setResult('resSavings', 'Stima non disponibile');

        setResult('resPayback', 'Non calcolabile');

        setResult('resScore', 'Serve valutazione FV + accumulo');

        setResult('detail1', "Senza un impianto fotovoltaico esistente, questo simulatore non fornisce una stima affidabile dell'accumulo standalone.");

        setResult('detail2', "Il massimo valore economico dell'accumulo emerge quando la batteria è abbinata a un impianto fotovoltaico con eccedenze da spostare.");

        setResult('detail3', 'Ti conviene fare una simulazione combinata FV + accumulo per ottenere un business case più credibile.');

        setResult('detail4', 'Hai inserito un consumo annuo di ' + formatNumber(consumoAnnuo, 0) + ' kWh e un costo energia di ' + formatCurrency(costoEnergia, 2) + '/kWh.');

        resultsGrid.style.display = 'grid';

        leadBox.style.display = 'block';

        resultsGrid.scrollIntoView({ behavior: 'smooth', block: 'start' });

        return;

      }


      const coeff = profileMap[profilo];

      const produzioneFV = fvPower * constants.produzionePerKwp;

      const autoconsumoDiretto = produzioneFV * coeff.autoconsumo;

      const eccedenzaFV = Math.max(produzioneFV - autoconsumoDiretto, 0);

      const energiaSpostataLorda = eccedenzaFV * coeff.utilizzo;

      const energiaSpostataNetta = energiaSpostataLorda * constants.rendimentoBatteria;

      const batteriaBase = fvPower * coeff.taglia;

      const batteriaArrotondata = Math.ceil(Math.max(batteriaBase, constants.batteriaMin) / constants.tagliaBloccoKwh) * constants.tagliaBloccoKwh;

      const batteriaKwh = Math.min(batteriaArrotondata, constants.batteriaMax);

      const valoreUnitario = constants.valoreUnitarioAccumulo;

      const risparmioEnergyShifting = energiaSpostataNetta * valoreUnitario;

      const bonusPeak = risparmioEnergyShifting * peakMap[picchiSelection.value];

      const risparmioTotale = risparmioEnergyShifting + bonusPeak;

      const numeroBlocchi = batteriaKwh / constants.tagliaBloccoKwh;

      const costoBatteria = numeroBlocchi * constants.costoPerBlocco100kwh;

      const payback = risparmioTotale > 0 ? costoBatteria / risparmioTotale : Infinity;

      const autoconsumoPercent = produzioneFV > 0 ? autoconsumoDiretto / produzioneFV * 100 : 0;

      const coperturaConsumi = consumoAnnuo > 0 ? energiaSpostataNetta / consumoAnnuo * 100 : 0;

      let potenzaText = 'Potenza impegnata non indicata: il beneficio da peak shaving è stimato in modo prudenziale.';

      if (isFinite(potenzaImpegnata) && potenzaImpegnata > 0) {

        potenzaText = 'Potenza impegnata indicata: ' + formatNumber(potenzaImpegnata, 0) + ' kW.';

      }


      setResult('resBattery', formatNumber(batteriaKwh, 0) + ' kWh');

      setResult('resSavings', formatCurrency(risparmioTotale, 0));

      setResult('resPayback', isFinite(payback) ? formatNumber(payback, 1) + ' anni' : 'Non calcolabile');

      setResult('resScore', getRating(payback));

      setResult('detail1', 'Produzione FV stimata: ' + formatNumber(produzioneFV, 0) + ' kWh/anno. Autoconsumo diretto stimato: ' + formatNumber(autoconsumoPercent, 0) + '%.');

      setResult('detail2', 'Energia annuale spostata dalla batteria: ' + formatNumber(energiaSpostataNetta, 0) + ' kWh. Valore unitario dell\'accumulo: ' + formatCurrency(valoreUnitario, 2) + '/kWh.');

      setResult('detail3', 'Costo indicativo del sistema di accumulo: ' + formatCurrency(costoBatteria, 0) + ' (' + formatNumber(numeroBlocchi, 0) + ' blocchi da 100 kWh). Copertura sui consumi annuali: ' + formatNumber(coperturaConsumi, 1) + '%.');

      setResult('detail4', potenzaText);

      resultsGrid.style.display = 'grid';

      leadBox.style.display = 'block';

      resultsGrid.scrollIntoView({ behavior: 'smooth', block: 'start' });

    });
